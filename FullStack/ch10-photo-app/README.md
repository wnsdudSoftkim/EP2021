## Add User

```

 --region ap-northeast-2 \
 --client-id 7nnkbshbkoalmv1uv3glfv0vhj \
 --username admin \
 --password Passw0rd! \
 --user-attributes Name="email",Value="admin@example.com"

aws cognito-idp admin-confirm-sign-up --profile fullstack \
 --region ap-northeast-2 \
 --user-pool-id ap-northeast-2_3LZtXlUQ7 \
 --username admin

$ yarn start
```
